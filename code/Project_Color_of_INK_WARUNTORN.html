# Create lyric list from INK WARUNTORN url
# Thanks เพลง.meemodel.com for data

ink_url = 'https://เพลง.meemodel.com/ศิลปิน/อิ้งค์%20วรันธร'
root_url = 'https://เพลง.meemodel.com'

response = requests.get(ink_url, verify = False)
soup = BeautifulSoup(response.text, 'html.parser')

tracks = soup.find_all('tr')
track_metadata = defaultdict(list)

for id, tr in tqdm(enumerate(tracks)):
  metadata = [x.text for x in tr.find_all('td')]
  url = [a.get('href') for a in tr.find_all('a')]

  if len(metadata) > 0 and len(url) > 0:   

    track_metadata['track_id'].append(id)
    track_metadata['title'].append(metadata[0])
    track_metadata['artist'].append(metadata[1])
    track_metadata['lyric_url'].append(''.join([root_url, url[0]]))

    lyric_response = requests.get(''.join([root_url, url[0]]), verify = False)
    lyric_html = BeautifulSoup(lyric_response.text, 'html.parser')

    track_metadata['lyric'].append(lyric_html.find('div', {'class': 'lyric'}).text)

ink_lyric = pd.DataFrame(track_metadata).set_index('track_id')
